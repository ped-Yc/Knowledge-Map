## 伪代码实现
```javascript
// 对数组A进行排序
SELECTION-SORT(A)
for i=0 to A.length-1
	key = A[i]
	j = i+1
	min = A[j]
	for(j to A.length)
		if(min>=A[j])
			min=A[j]
			idx=j
	if(A[i]>min)
		A[i] = min
		A[idx] = key		
```

## 代码实现
```javascript
    function SELECTION_SORT(A) {
      for (let i = 0; i < A.length - 1; i++) {
	    let idx;
        let key = A[i];
        let j = i + 1;
        let min = A[j];
        for (j; j < A.length; j++) {
          if (min >= A[j]) {
            min = A[j];
            idx = j;
          }
        }
        if (A[i] > min) {
          A[i] = min;
          A[idx] = key;
        }
      }
    }
```
